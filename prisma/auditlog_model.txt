
model AuditLog {
  id         String   @id @default(cuid())
  userId     String
  userEmail  String?
  userRole   String
  action     String
  entityType String
  entityId   String
  details    String?
  createdAt  DateTime @default(now())

  @@index([userId])
  @@index([action])
  @@index([entityType])
  @@index([createdAt])
}
